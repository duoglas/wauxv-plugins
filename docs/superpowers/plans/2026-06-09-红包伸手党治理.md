// [PROBE v-temp] 仅探测: 枚举 el 的 long/Long 字段, 打印 字段名+值是否落 [now-2天, now+1分] epoch 区间(秒&毫秒两种判定), 不打印金额/昵称/wxid。
try {
    long nowS = System.currentTimeMillis() / 1000L, nowMs = System.currentTimeMillis();
    java.lang.reflect.Field[] fs = el.getClass().getDeclaredFields();
    StringBuilder pb = new StringBuilder("[RP][PROBE] fields: ");
    for (int fi = 0; fi < fs.length; fi++) {
        fs[fi].setAccessible(true);
        Object fv = null; try { fv = fs[fi].get(el); } catch (Throwable t) {}
        if (fv instanceof Long || (fv != null && fv.getClass() == long.class)) {
            long lv = ((Long) fv).longValue();
            boolean asSec = lv > (nowS - 172800) && lv < (nowS + 60);
            boolean asMs  = lv > (nowMs - 172800000L) && lv < (nowMs + 60000L);
            pb.append(fs[fi].getName()).append("=").append(asSec ? "SEC?" : (asMs ? "MS?" : "long")).append("; ");
        }
    }
    hbBgLog(pb.toString());
} catch (Throwable t) {}
